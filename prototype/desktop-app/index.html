// useStatusBarData.ts
export function useStatusBarData() {
  const stats = useStats()
  const model = useCurrentModel()

  const contextPercent = useMemo(() => {
    const used = stats.inputTokens + stats.outputTokens
    return Math.round((used / model.contextWindow) * 100)
  }, [stats, model])

  const progressColor = getProgressColor(contextPercent)
  const cost = formatCost(stats.totalCost)

  return { model, stats, contextPercent, progressColor, cost }
}
